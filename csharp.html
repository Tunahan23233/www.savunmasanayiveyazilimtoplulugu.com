        private void button8_Click(object sender, EventArgs e)
        {
            
            for(int i=0;i<sayilarxy.GetLength(0);i++)
            {
                string kelime = "";
                for (int j = 0; j < sayilarxy.GetLength(1); j++)
                {
                   kelime=sayilarxy[i,j].ToString()+" "+kelime;
                }
                listBox1.Items.Add(kelime);
            }
        }

        private void button9_Click(object sender, EventArgs e)
        {
            for (int i = 0; i < sayilarxy.GetLength(0); i++)
            {
                string kelime = "";
                for (int j = sayilarxy.GetLength(1)-1; j >= 0; j--)
                {
                    kelime = sayilarxy[i, j].ToString() + " " + kelime;
                }
                listBox1.Items.Add(kelime);
            }
        }
    }
